
-- affiliate_products --
CREATE TABLE `affiliate_products` (

  `main_image_url_snapshot` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL COMMENT 'Snapshot of main image URL (from Amazon)',
  `try_on_image_url` text CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci,
  `item_type` text CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci,
  `is_featured` tinyint(1) DEFAULT NULL,
  `status` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci NOT NULL DEFAULT 'active',
  `category_id` bigint unsigned DEFAULT NULL,
  `local_image_path` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `created_at` timestamp NULL DEFAULT NULL,
  `updated_at` timestamp NULL DEFAULT NULL,
  `try_on_type` varchar(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL COMMENT 'e.g., earring, necklace, ring',
  `try_on_anchor` varchar(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci DEFAULT NULL COMMENT 'AI anchor point: e.g., ear_tragion_left, chin_center, ring_finger_mcp',
  `try_on_scale_default` decimal(5,3) DEFAULT '0.100' COMMENT 'Default scale for the item (0.01 to 1.0)',
  `try_on_offset_y` int DEFAULT '0' COMMENT 'Vertical pixel offset from anchor',
  `seo_title` varchar(255) COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `meta_description` varchar(255) COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `og_title` varchar(255) COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `og_description` varchar(255) COLLATE utf8mb4_unicode_ci DEFAULT NULL,
  `schema_type` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL DEFAULT 'Product',
  PRIMARY KEY (`id`),
  UNIQUE KEY `affiliate_products_amazon_asin_unique` (`amazon_asin`),
  KEY `affiliate_products_category_id_foreign` (`category_id`),
  CONSTRAINT `affiliate_products_category_id_foreign` FOREIGN KEY (`category_id`) REFERENCES `categories` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB AUTO_INCREMENT=88 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci

-- diamond_pricings --
CREATE TABLE `diamond_pricings` (
  `id` int NOT NULL AUTO_INCREMENT,
  `carat_range_min` decimal(3,2) NOT NULL,
  `carat_range_max` decimal(3,2) NOT NULL,
  `cut_grade` enum('Excellent','Very Good','Good','Fair','Poor') NOT NULL,
  `color_grade` enum('D','E','F','G','H','I','J','K','L','M','N') NOT NULL,
  `clarity_grade` enum('FL','IF','VVS1','VVS2','VS1','VS2','SI1','SI2','I1','I2','I3') NOT NULL,
  `base_price_per_carat` decimal(10,2) NOT NULL,
  `price_multiplier` decimal(4,3) DEFAULT '1.000',
  `shape` varchar(20) DEFAULT 'Round',
  `fluorescence` enum('None','Faint','Medium','Strong') DEFAULT 'None',
  `certification` varchar(10) DEFAULT 'GIA',
  `date_updated` date NOT NULL,
  `market_region` varchar(10) DEFAULT 'Global',
  `is_active` tinyint(1) DEFAULT '1',
  `created_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_carat_cut_color_clarity` (`carat_range_min`,`carat_range_max`,`cut_grade`,`color_grade`,`clarity_grade`),
  KEY `idx_date_updated` (`date_updated`)
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
